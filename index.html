  <!-- Draggable Window Script & System Logic -->

  <script>

    // System Windows State

    const windowIds = ['window-about', 'window-portfolio', 'window-universe', 'window-contact'];

    let zIndexCounter = 50;


    // Clock System

    function updateClock() {

      const clockElement = document.getElementById('system-clock');

      if (clockElement) {

        const now = new Date();

        clockElement.textContent = now.toLocaleTimeString([], { hour: '2-digit', minute: '2-digit', second: '2-digit' });

      }

    }

    setInterval(updateClock, 1000);

    updateClock();


    // Toggle Retro Menu

    function toggleMenu() {

      const startMenu = document.getElementById('start-menu');

      startMenu.classList.toggle('hidden');

    }


    // Close Menu if clicked outside

    document.addEventListener('click', function(e) {

      const menuBtn = document.getElementById('menu-btn');

      const startMenu = document.getElementById('start-menu');

      if (!menuBtn.contains(e.target) && !startMenu.contains(e.target)) {

        startMenu.classList.add('hidden');

      }

    });


    // Window Management Functions

    function openWindow(id) {

      const win = document.getElementById(id);

      if (win) {

        win.classList.add('active');

        focusWindow(win);

        updateTaskbar();

      }

    }


    function menuOpenWindow(id) {

      openWindow(id);

      document.getElementById('start-menu').classList.add('hidden');

    }


    function closeWindow(id) {

      const win = document.getElementById(id);

      if (win) {

        win.classList.remove('active');

        updateTaskbar();

      }

    }


    // Toggle minimize state

    function minimizeWindow(id) {

      const win = document.getElementById(id);

      if (win) {

        win.classList.remove('active');

        updateTaskbar();

      }

    }


    function toggleMaximize(id) {

      const win = document.getElementById(id);

      if (win) {

        win.classList.toggle('maximized');

      }

    }


    function focusWindow(win) {

      // Remove focus from all windows

      document.querySelectorAll('.window').forEach(w => w.classList.remove('focused'));

      // Increase zIndex counter and apply to focused window

      zIndexCounter += 1;

      win.style.zIndex = zIndexCounter;

      win.classList.add('focused');

    }


    // Setup Window Events (Focus and Drag)

    document.querySelectorAll('.window').forEach(win => {

      // Focus on click

      win.addEventListener('pointerdown', () => {

        focusWindow(win);

      });


      // Draggable Header Setup

      const header = win.querySelector('.window-header');

      if (header) {

        let isDragging = false;

        let startX, startY;

        let originalLeft, originalTop;


        header.addEventListener('pointerdown', (e) => {

          // Prevent drag if maximize window buttons or close buttons are clicked

          if (e.target.classList.contains('window-btn')) return;

          // Prevent dragging if window is maximized

          if (win.classList.contains('maximized')) return;


          isDragging = true;

          focusWindow(win);

          

          // Get current window coordinates

          const style = window.getComputedStyle(win);

          originalLeft = parseFloat(style.left) || 0;

          originalTop = parseFloat(style.top) || 0;


          startX = e.clientX;

          startY = e.clientY;


          header.setPointerCapture(e.pointerId);

          e.preventDefault();

        });


        header.addEventListener('pointermove', (e) => {

          if (!isDragging) return;

          const deltaX = e.clientX - startX;

          const deltaY = e.clientY - startY;


          // Apply boundaries check roughly

          let newLeft = originalLeft + deltaX;

          let newTop = originalTop + deltaY;


          // Keep within reasonable viewport limits

          if (newTop < 0) newTop = 0;

          if (newTop > window.innerHeight - 80) newTop = window.innerHeight - 80;


          win.style.left = `${newLeft}px`;

          win.style.top = `${newTop}px`;

        });


        const stopDragging = (e) => {

          if (isDragging) {

            header.releasePointerCapture(e.pointerId);

            isDragging = false;

          }

        };


        header.addEventListener('pointerup', stopDragging);

        header.addEventListener('pointercancel', stopDragging);


        // Double click to maximize

        header.addEventListener('dblclick', (e) => {

          if (e.target.classList.contains('window-btn')) return;

          toggleMaximize(win.id);

        });

      }

    });


    // Update Bottom Taskbar Dynamic Items

    function updateTaskbar() {

      const container = document.getElementById('taskbar-tasks');

      if (!container) return;

      container.innerHTML = '';


      windowIds.forEach(id => {

        const win = document.getElementById(id);

        if (win && win.classList.contains('active')) {

          const title = win.querySelector('.window-header span').textContent.trim();

          

          // Create task button

          const btn = document.createElement('button');

          btn.className = `px-3 py-1 border-retro font-mono text-[10px] font-bold uppercase tracking-wider flex items-center space-x-1 shadow-retro-sm transition-all focus:outline-none ${win.classList.contains('focused') ? 'bg-pink' : 'bg-creamdark'}`;

          btn.innerHTML = `<span>📂</span> <span>${title.toLowerCase()}</span>`;

          btn.onclick = () => {

            if (win.classList.contains('focused')) {

              minimizeWindow(id);

            } else {

              openWindow(id);

            }

          };

          container.appendChild(btn);

        }

      });

    }


    // Portfolio Tabs Switcher

    function switchTab(tabName) {

      // Hide all content blocks

      document.querySelectorAll('.portfolio-tab-content').forEach(el => el.classList.add('hidden'));

      // Show matching block

      document.getElementById(`tab-${tabName}`).classList.remove('hidden');


      // Update tabs UI states

      document.querySelectorAll('.portfolio-tab').forEach(el => {

        el.classList.add('bg-creamdark', 'opacity-60');

        el.classList.remove('bg-cream');

      });

      const activeTab = document.getElementById(`tab-btn-${tabName}`);

      activeTab.classList.remove('bg-creamdark', 'opacity-60');

      activeTab.classList.add('bg-cream');

    }


    // Form Submission Handling

    function handleFormSubmit(e) {

      e.preventDefault();

      const name = document.getElementById('name').value;

      const email = document.getElementById('email').value;

      const message = document.getElementById('message').value;


      // Animate postcard success action

      alert(`Thank you, ${name}! Your postcard is stamped and sent to the digital atelier archive.`);

      e.target.reset();

      closeWindow('window-contact');

    }


    // Initialize desktop layout configurations

    window.addEventListener('DOMContentLoaded', () => {

      // Adjust default window coordinates based on screen sizes

      const isMobile = window.innerWidth < 768;

      if (isMobile) {

        document.querySelectorAll('.window').forEach(win => {

          win.style.left = '2.5vw';

          win.style.top = '10%';

          win.style.width = '95vw';

        });

      }

      updateTaskbar();

    });

  </script>
